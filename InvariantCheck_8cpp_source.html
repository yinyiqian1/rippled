44bool
45TransactionFeeCheck::finalize(
46 STTx const& tx,
47 TER const,
48 XRPAmount const fee,
49 ReadView const&,
50 beast::Journal const& j)
51{
52 // We should never charge a negative fee
53 if (fee.drops() < 0)
54 {
55 JLOG(j.fatal()) << "Invariant failed: fee paid was negative: "
56 << fee.drops();
57 return false;
58 }
59
60 // We should never charge a fee that's greater than or equal to the
61 // entire XRP supply.
62 if (fee >= INITIAL_XRP)
63 {
64 JLOG(j.fatal()) << "Invariant failed: fee paid exceeds system limit: "
65 << fee.drops();
66 return false;
67 }
68
69 // We should never charge more for a transaction than the transaction
70 // authorizes. It's possible to charge less in some circumstances.
71 if (fee > tx.getFieldAmount(sfFee).xrp())
72 {
73 JLOG(j.fatal()) << "Invariant failed: fee paid is " << fee.drops()
74 << " exceeds fee specified in transaction.";
75 return false;
76 }
77
78 return true;
79}

83void
84XRPNotCreated::visitEntry(
85 bool isDelete,
86 std::shared_ptr<SLE const> const& before,
87 std::shared_ptr<SLE const> const& after)
88{
89 /* We go through all modified ledger entries, looking only at account roots,
90 * escrow payments, and payment channels. We remove from the total any
91 * previous XRP values and add to the total any new XRP values. The net
92 * balance of a payment channel is computed from two fields (amount and
93 * balance) and deletions are ignored for paychan and escrow because the
94 * amount fields have not been adjusted for those in the case of deletion.
95 */
96 if (before)
97 {
98 switch (before->getType())
99 {
100 case ltACCOUNT_ROOT:
101 drops_ -= (*before)[sfBalance].xrp().drops();
102 break;
103 case ltPAYCHAN:
104 drops_ -=
105 ((*before)[sfAmount] - (*before)[sfBalance]).xrp().drops();
106 break;
107 case ltESCROW:
108 drops_ -= (*before)[sfAmount].xrp().drops();
109 break;
110 default:
111 break;
112 }
113 }
114
115 if (after)
116 {
117 switch (after->getType())
118 {
119 case ltACCOUNT_ROOT:
120 drops_ += (*after)[sfBalance].xrp().drops();
121 break;
122 case ltPAYCHAN:
123 if (!isDelete)
124 drops_ += ((*after)[sfAmount] - (*after)[sfBalance])
125 .xrp()
126 .drops();
127 break;
128 case ltESCROW:
129 if (!isDelete)
130 drops_ += (*after)[sfAmount].xrp().drops();
131 break;
132 default:
133 break;
134 }
135 }
136}

138bool
139XRPNotCreated::finalize(
140 STTx const& tx,
141 TER const,
142 XRPAmount const fee,
143 ReadView const&,
144 beast::Journal const& j)
145{
146 // The net change should never be positive, as this would mean that the
147 // transaction created XRP out of thin air. That's not possible.
148 if (drops_ > 0)
149 {
150 JLOG(j.fatal()) << "Invariant failed: XRP net change was positive: "
151 << drops_;
152 return false;
153 }
154
155 // The negative of the net change should be equal to actual fee charged.
156 if (-drops_ != fee.drops())
157 {
158 JLOG(j.fatal()) << "Invariant failed: XRP net change of " << drops_
159 << " doesn't match fee " << fee.drops();
160 return false;
161 }
162
163 return true;
164}

168void
169XRPBalanceChecks::visitEntry(
170 bool,
171 std::shared_ptr<SLE const> const& before,
172 std::shared_ptr<SLE const> const& after)
173{
174 auto isBad = [](STAmount const& balance) {
175 if (!balance.native())
176 return true;
177
178 auto const drops = balance.xrp();
179
180 // Can't have more than the number of drops instantiated
181 // in the genesis ledger.
182 if (drops > INITIAL_XRP)
183 return true;
184
185 // Can't have a negative balance (0 is OK)
186 if (drops < XRPAmount{0})
187 return true;
188
189 return false;
190 };
191
192 if (before && before->getType() == ltACCOUNT_ROOT)
193 bad_ |= isBad((*before)[sfBalance]);
194
195 if (after && after->getType() == ltACCOUNT_ROOT)
196 bad_ |= isBad((*after)[sfBalance]);
197}

223{

238
239 if (after && after->getType() == ltOFFER)
240 bad_ |= isBad((*after)[sfTakerPays], (*after)[sfTakerGets]);

317bool
318AccountRootsNotDeleted::finalize(
319 STTx const& tx,
320 TER const result,
321 XRPAmount const,
322 ReadView const&,
323 beast::Journal const& j)
324{
325 // AMM account root can be deleted as the result of AMM withdraw/delete
326 // transaction when the total AMM LP Tokens balance goes to 0.
327 // A successful AccountDelete or AMMDelete MUST delete exactly
328 // one account root.
329 if ((tx.getTxnType() == ttACCOUNT_DELETE ||
330 tx.getTxnType() == ttAMM_DELETE) &&
331 result == tesSUCCESS)
332 {
333 if (accountsDeleted_ == 1)
334 return true;
335
336 if (accountsDeleted_ == 0)
337 JLOG(j.fatal()) << "Invariant failed: account deletion "
338 "succeeded without deleting an account";
339 else
340 JLOG(j.fatal()) << "Invariant failed: account deletion "
341 "succeeded but deleted multiple accounts!";
342 return false;
343 }
344
345 // A successful AMMWithdraw MAY delete one account root
346 // when the total AMM LP Tokens balance goes to 0. Not every AMM withdraw
347 // deletes the AMM account, accountsDeleted_ is set if it is deleted.
348 if (tx.getTxnType() == ttAMM_WITHDRAW && result == tesSUCCESS &&
349 accountsDeleted_ == 1)
350 return true;
351
352 if (accountsDeleted_ == 0)
353 return true;
354
355 JLOG(j.fatal()) << "Invariant failed: an account root was deleted";
356 return false;
357}

371bool
372AccountRootsDeletedClean::finalize(
373 STTx const& tx,
374 TER const result,
375 XRPAmount const,
376 ReadView const& view,
377 beast::Journal const& j)
378{
379 // Always check for objects in the ledger, but to prevent differing
380 // transaction processing results, however unlikely, only fail if the
381 // feature is enabled. Enabled, or not, though, a fatal-level message will
382 // be logged
383 bool const enforce = view.rules().enabled(featureInvariantsV1_1);
384
385 auto const objectExists = [&view, enforce, &j](auto const& keylet) {
386 if (auto const sle = view.read(keylet))
387 {
388 // Finding the object is bad
389 auto const typeName = [&sle]() {
390 auto item =
391 LedgerFormats::getInstance().findByType(sle->getType());
392
393 if (item != nullptr)
394 return item->getName();
395 return std::to_string(sle->getType());
396 }();
397
398 JLOG(j.fatal())
399 << "Invariant failed: account deletion left behind a "
400 << typeName << " object";
401 (void)enforce;
402 assert(enforce);
403 return true;
404 }
405 return false;
406 };
407
408 for (auto const& accountSLE : accountsDeleted_)
409 {
410 auto const accountID = accountSLE->getAccountID(sfAccount);
411 // Simple types
412 for (auto const& [keyletfunc, _, __] : directAccountKeylets)
413 {
414 if (objectExists(std::invoke(keyletfunc, accountID)) && enforce)
415 return false;
416 }
417
418 {
419 // NFT pages. ntfpage_min and nftpage_max were already explicitly
420 // checked above as entries in directAccountKeylets. This uses
421 // view.succ() to check for any NFT pages in between the two
422 // endpoints.
423 Keylet const first = keylet::nftpage_min(accountID);
424 Keylet const last = keylet::nftpage_max(accountID);
425
426 std::optional<uint256> key = view.succ(first.key, last.key.next());
427
428 // current page
429 if (key && objectExists(Keylet{ltNFTOKEN_PAGE, *key}) && enforce)
430 return false;
431 }
432
433 // Keys directly stored in the AccountRoot object
434 if (auto const ammKey = accountSLE->at(~sfAMMID))
435 {
436 if (objectExists(keylet::amm(*ammKey)) && enforce)
437 return false;
438 }
439 }
440
441 return true;
442}

562bool
563ValidNewAccountRoot::finalize(
564 STTx const& tx,
565 TER const result,
566 XRPAmount const,
567 ReadView const& view,
568 beast::Journal const& j)
569{
570 if (accountsCreated_ == 0)
571 return true;
572
573 if (accountsCreated_ > 1)
574 {
575 JLOG(j.fatal()) << "Invariant failed: multiple accounts "
576 "created in a single transaction";
577 return false;
578 }
579
580 // From this point on we know exactly one account was created.
581 if ((tx.getTxnType() == ttPAYMENT || tx.getTxnType() == ttAMM_CREATE ||
582 tx.getTxnType() == ttXCHAIN_ADD_CLAIM_ATTESTATION ||
583 tx.getTxnType() == ttXCHAIN_ADD_ACCOUNT_CREATE_ATTESTATION) &&
584 result == tesSUCCESS)
585 {
586 std::uint32_t const startingSeq{
587 view.rules().enabled(featureDeletableAccounts) ? view.seq() : 1};
588
589 if (accountSeq_ != startingSeq)
590 {
591 JLOG(j.fatal()) << "Invariant failed: account created with "
592 "wrong starting sequence number";
593 return false;
594 }
595 return true;
596 }
597
598 JLOG(j.fatal()) << "Invariant failed: account root created "
599 "by a non-Payment, by an unsuccessful transaction, "
600 "or by AMM";
601 return false;
602}

606void
607ValidNFTokenPage::visitEntry(
608 bool isDelete,
609 std::shared_ptr<SLE const> const& before,
610 std::shared_ptr<SLE const> const& after)
611{
612 static constexpr uint256 const& pageBits = nft::pageMask;
613 static constexpr uint256 const accountBits = ~pageBits;
614
615 if ((before && before->getType() != ltNFTOKEN_PAGE) ||
616 (after && after->getType() != ltNFTOKEN_PAGE))
617 return;
618
619 auto check = [this, isDelete](std::shared_ptr<SLE const> const& sle) {
620 uint256 const account = sle->key() & accountBits;
621 uint256 const hiLimit = sle->key() & pageBits;
622 std::optional<uint256> const prev = (*sle)[~sfPreviousPageMin];
623
624 // Make sure that any page links...
625 // 1. Are properly associated with the owning account and
626 // 2. The page is correctly ordered between links.
627 if (prev)
628 {
629 if (account != (*prev & accountBits))
630 badLink_ = true;
631
632 if (hiLimit <= (*prev & pageBits))
633 badLink_ = true;
634 }
635
636 if (auto const next = (*sle)[~sfNextPageMin])
637 {
638 if (account != (*next & accountBits))
639 badLink_ = true;
640
641 if (hiLimit >= (*next & pageBits))
642 badLink_ = true;
643 }
644
645 {
646 auto const& nftokens = sle->getFieldArray(sfNFTokens);
647
648 // An NFTokenPage should never contain too many tokens or be empty.
649 if (std::size_t const nftokenCount = nftokens.size();
650 (!isDelete && nftokenCount == 0) ||
651 nftokenCount > dirMaxTokensPerPage)
652 invalidSize_ = true;
653
654 // If prev is valid, use it to establish a lower bound for
655 // page entries. If prev is not valid the lower bound is zero.
656 uint256 const loLimit =
657 prev ? *prev & pageBits : uint256(beast::zero);
658
659 // Also verify that all NFTokenIDs in the page are sorted.
660 uint256 loCmp = loLimit;
661 for (auto const& obj : nftokens)
662 {
663 uint256 const tokenID = obj[sfNFTokenID];
664 if (!nft::compareTokens(loCmp, tokenID))
665 badSort_ = true;
666 loCmp = tokenID;
667
668 // None of the NFTs on this page should belong on lower or
669 // higher pages.
670 if (uint256 const tokenPageBits = tokenID & pageBits;
671 tokenPageBits < loLimit || tokenPageBits >= hiLimit)
672 badEntry_ = true;
673
674 if (auto uri = obj[~sfURI]; uri && uri->empty())
675 badURI_ = true;
676 }
677 }
678 };
679
680 if (before)
681 {
682 check(before);
683
684 // While an account's NFToken directory contains any NFTokens, the last
685 // NFTokenPage (with 96 bits of 1 in the low part of the index) should
686 // never be deleted.
687 if (isDelete && (before->key() & nft::pageMask) == nft::pageMask &&
688 before->isFieldPresent(sfPreviousPageMin))
689 {
690 deletedFinalPage_ = true;
691 }
692 }
693
694 if (after)
695 check(after);
696
697 if (!isDelete && before && after)
698 {
699 // If the NFTokenPage
700 // 1. Has a NextMinPage field in before, but loses it in after, and
701 // 2. This is not the last page in the directory
702 // Then we have identified a corruption in the links between the
703 // NFToken pages in the NFToken directory.
704 if ((before->key() & nft::pageMask) != nft::pageMask &&
705 before->isFieldPresent(sfNextPageMin) &&
706 !after->isFieldPresent(sfNextPageMin))
707 {
708 deletedLink_ = true;
709 }
710 }
711}

713bool
714ValidNFTokenPage::finalize(
715 STTx const& tx,
716 TER const result,
717 XRPAmount const,
718 ReadView const& view,
719 beast::Journal const& j)
720{
721 if (badLink_)
722 {
723 JLOG(j.fatal()) << "Invariant failed: NFT page is improperly linked.";
724 return false;

726
727 if (badEntry_)

729 JLOG(j.fatal()) << "Invariant failed: NFT found in incorrect page.";
730 return false;
731 }
732

736 return false;
737 }
738
739 if (badURI_)
740 {
741 JLOG(j.fatal()) << "Invariant failed: NFT contains empty URI.";
742 return false;
743 }

747 JLOG(j.fatal()) << "Invariant failed: NFT page has invalid size.";
748 return false;
749 }
750

752 {
753 if (deletedFinalPage_)
754 {
755 JLOG(j.fatal()) << "Invariant failed: Last NFT page deleted with "
756 "non-empty directory.";
757 return false;
758 }
759 if (deletedLink_)
760 {
761 JLOG(j.fatal()) << "Invariant failed: Lost NextMinPage link.";
762 return false;
763 }
764 }
765
766 return true;
767}

771NFTokenCountTracking::visitEntry(

778 beforeMintedTotal += (*before)[~sfMintedNFTokens].value_or(0);
779 beforeBurnedTotal += (*before)[~sfBurnedNFTokens].value_or(0);
780 }
781
782 if (after && after->getType() == ltACCOUNT_ROOT)
783 {
784 afterMintedTotal += (*after)[~sfMintedNFTokens].value_or(0);
785 afterBurnedTotal += (*after)[~sfBurnedNFTokens].value_or(0);

788
789bool
790NFTokenCountTracking::finalize(
791 STTx const& tx,
792 TER const result,
793 XRPAmount const,
794 ReadView const& view,
795 beast::Journal const& j)
796{
797 if (TxType const txType = tx.getTxnType();
798 txType != ttNFTOKEN_MINT && txType != ttNFTOKEN_BURN)
799 {
800 if (beforeMintedTotal != afterMintedTotal)
801 {
802 JLOG(j.fatal()) << "Invariant failed: the number of minted tokens "
803 "changed without a mint transaction!";
804 return false;
805 }
806
807 if (beforeBurnedTotal != afterBurnedTotal)
808 {
809 JLOG(j.fatal()) << "Invariant failed: the number of burned tokens "
810 "changed without a burn transaction!";
811 return false;
812 }
813
814 return true;
815 }
816
817 if (tx.getTxnType() == ttNFTOKEN_MINT)
818 {
819 if (result == tesSUCCESS && beforeMintedTotal >= afterMintedTotal)
820 {
821 JLOG(j.fatal())
822 << "Invariant failed: successful minting didn't increase "
823 "the number of minted tokens.";
824 return false;
825 }
826
827 if (result != tesSUCCESS && beforeMintedTotal != afterMintedTotal)
828 {
829 JLOG(j.fatal()) << "Invariant failed: failed minting changed the "
830 "number of minted tokens.";
831 return false;
832 }
833
834 if (beforeBurnedTotal != afterBurnedTotal)
835 {
836 JLOG(j.fatal())
837 << "Invariant failed: minting changed the number of "
838 "burned tokens.";
839 return false;
840 }
841 }
842
843 if (tx.getTxnType() == ttNFTOKEN_BURN)
844 {
845 if (result == tesSUCCESS)
846 {
847 if (beforeBurnedTotal >= afterBurnedTotal)
848 {
849 JLOG(j.fatal())

852 return false;
853 }

891 XRPAmount const,

894{
895 if (tx.getTxnType() != ttCLAWBACK)
896 return true;
897

899 {
900 if (trustlinesChanged > 1)
901 {
902 JLOG(j.fatal())

904 return false;
905 }
906
907 AccountID const issuer = tx.getAccountID(sfAccount);
908 STAmount const amount = tx.getFieldAmount(sfAmount);
909 AccountID const& holder = amount.getIssuer();

912

914 {
915 JLOG(j.fatal())
916 << "Invariant failed: trustline balance is negative";
917 return false;
918 }
919 }
920 else
921 {
922 if (trustlinesChanged != 0)
923 {
924 JLOG(j.fatal()) << "Invariant failed: some trustlines were changed "
925 "despite failure of the transaction.";
926 return false;
927 }
928 }

930 return true;
